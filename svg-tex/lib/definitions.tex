\tikzset{
  pipe dream cell/.style={draw=black!35, line width=0.35pt, fill=white},
  pipe dream dot/.style={fill=black!45},
  pipe dream cross/.style={draw=black, line width=0.7pt}
}

% Draw a staircase-shaped pipe dream grid.
%
% Usage:
%   \pipeDreamStaircase[<tikz options>]{n}{row/col,row/col,...}
%
% The coordinate list gives the cells containing crosses.  Rows and columns are
% one-indexed, matching the notation D(a,\alpha)=\{(\alpha_j,a_j-\alpha_j+1)\}.
\newcommand{\pipeDreamStaircase}[3][]{%
  \begin{tikzpicture}[
      x=0.55cm,
      y=0.55cm,
      line cap=round,
      line join=round,
      #1
    ]
    \pgfmathtruncatemacro{\pdn}{#2}
    \foreach \row in {1,...,\pdn} {
      \pgfmathtruncatemacro{\pdlast}{\pdn-\row+1}
      \foreach \col in {1,...,\pdlast} {
        \draw[pipe dream cell] (\col-1,-\row+1) rectangle (\col,-\row);
        \fill[pipe dream dot] (\col-0.5,-\row+0.5) circle (0.45pt);
      }
    }
    \foreach \row/\col in {#3} {
      \draw[pipe dream cross] (\col-0.68,-\row+0.5) -- (\col-0.32,-\row+0.5);
      \draw[pipe dream cross] (\col-0.5,-\row+0.68) -- (\col-0.5,-\row+0.32);
    }
  \end{tikzpicture}%
}
