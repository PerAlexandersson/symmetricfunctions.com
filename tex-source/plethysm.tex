\metadescription{Definition of plethysm and some properties.}


\section[plethysm]{Plethysm}


See \cite{qtCatalanBook} and \cite{LoehrRemmel2010} for an introduction to calculations with plethysm.
Another good reference is \name{Mike Zabrocki}'s introduction to symmetric
functions \cite[Ch.~III]{Zabrocki2003}.



\subsection[plethysmGLn]{Representation-theoretic interpretation}

Plethysm can easily be described using \hyperref[representationGln]{representation theory} of $\GL_n$.

Let $U,V,W$ be vector spaces over $\setC$, and consider
polynomial representations $\phi : \GL(U) \to \GL(V)$ and $\psi : \GL(V) \to \GL(W)$.
Let $f$ and $g$ be the corresponding characters (symmetric functions).
The composition $\psi\circ \phi : \GL(U) \to \GL(W)$ is then also a polynomial representation,
and $g[f]$ is the \defin{plethysm}\label{plethysmOperation} of $g$ and $f$.

\name{Rosa Orellana}, \name{Franco Saliola}, \name{Anne Schilling}, and
\name{Mike Zabrocki} study plethysm through the uniform block permutation
algebra, also called the
\defin{party algebra}\label{partyAlgebra}
\cite{OrellanaSaliolaSchillingZabrocki2022Party}.
They define a Frobenius characteristic map whose character theory naturally
produces plethysms of Schur functions.

This can be made more concrete, and extended to any symmetric function, see below.

\subsection[plethysmDefinition]{Definition}

Plethysm is closely related to the \hyperref[powerSum]{power-sum symmetric functions}.
The following properties uniquely define plethysm, where $f$, $g$, and $h$
denote symmetric functions with \emph{integer} coefficients.
Brackets are commonly used to denote
\defin{plethystic substitutions}\label{plethysticSubstitution}.
\begin{itemize}
\item $\powerSum_k[\powerSum_m] = \powerSum_{km}$.
\item $\powerSum_k[f\pm g] = \powerSum_k[f] \pm \powerSum_k[g]$.
\item $\powerSum_k[f\cdot g] = \powerSum_k[f] \cdot \powerSum_k[g]$.
\item $(f\pm g)[h] = f[h] \pm g[h]$.
\item $(f\cdot g)[h] = f[h] \cdot g[h]$.
\end{itemize}
Since every symmetric function can be expressed as a sum of products of
$\powerSum_k(\xvec)$, these properties uniquely determine the expression
$f[g]$.

The operator sending $f(\xvec)$ to
$f(\xvec^k) = f[\powerSum_k(\xvec)]$ is called the
$k$-th \defin{Adams operator}\label{adamsOperator}.
The adjoint to the Adams operator is the
\defin{Verschiebung operator}\label{verschiebungOperator} $V_k$,
which is defined as $V_k(\completeH_r) = \completeH_{r/k}$, if $k \mid r$,
and $0$ otherwise.


\begin{example}
Let us compute $\powerSum_{22}[\powerSum_{43}]$:
\begin{align*}
\powerSum_{22}[\powerSum_{43}] &= (\powerSum_{2}[\powerSum_{43}])^2 \\
&= (\powerSum_{2}[\powerSum_{4}] \cdot \powerSum_{2}[\powerSum_{3}])^2 \\
&= (\powerSum_{8} \cdot \powerSum_{6})^2 \\
&= \powerSum_{8866}.
\end{align*}

Similarly, $\powerSum_{32}[3\powerSum_{4}]$ can be computed as
\begin{align*}
\powerSum_{32}[3\powerSum_{4}]
  &= (\powerSum_{3}[\powerSum_{4}+\powerSum_{4}+\powerSum_{4}])
     (\powerSum_{2}[\powerSum_{4}+\powerSum_{4}+\powerSum_{4}]) \\
&= (3\powerSum_{3}[\powerSum_{4}])(3\powerSum_{2}[\powerSum_{4}]) \\
&= 9\powerSum_{12} \cdot \powerSum_{8}.
\end{align*}
\end{example}


The situation is a bit more involved if the coefficients of $f$ and $g$
in the plethysm $f[g]$ are formal power series in, say, $\setC(\qvec)$.

We first define the plethysm $\powerSum_k[g(\xvec)]$.  If
$g(\xvec) = \sum_{\mu} d_\mu(\qvec) \powerSum_\mu(\xvec)$, then
\[
\powerSum_k[g(\xvec)] \coloneqq
\sum_{\mu} d_\mu(q_1^k,q_2^k,\dotsc) \; \powerSum_{k\mu}(\xvec) .
\]
Let now $f(\xvec) = \sum_{\lambda} c_\lambda(\qvec) \powerSum_\lambda(\xvec)$.
Then
\[
f[g(\xvec)] \coloneqq
\sum_{\lambda} c_\lambda(\qvec)
\prod_{j=1}^{\length(\lambda)} \powerSum_{\lambda_j}[g(\xvec)].
\]
This definition agrees with the previous one in the case $f$ and $g$ are symmetric
functions with integer coefficients, not depending on formal parameters.


\begin{example}
For example,
\[
\powerSum_{k}[5q \cdot \powerSum_{m}] = 5 q^k \powerSum_{km}.
\]
\end{example}

Capital letters are sometimes used to denote the sum of the variables in that alphabet.
For example, $X = x_1+x_2+\dotsb = \elementaryE_1(\xvec)$.

\name{Alessandro D'Andrea}, \name{Enrico Fatighenti}, and
\name{Claudio Onorati} study a derivation $\mathsf{D}$ on $\spaceSym$ and
prove a chain rule for $\mathsf{D}$ with respect to plethysm
\cite{DAndreaFatighentiOnorati2025x}.  They also relate the Schur-support
geometry of $\mathsf{D}(f)$ to that of a homogeneous symmetric function $f$.


\subsection[plethysmIdentities]{Identities}

We have the following identities:
\begin{itemize}
\item $\powerSum_{k}[qX] = q^k\powerSum_{k}[X]$
\item $\powerSum_{k}[-X] = -\powerSum_{k}[X]$
\item $\powerSum_{k}[\epsilon X] = \epsilon^k \powerSum_{k}[X]$ when $\epsilon=-1$.
\item $\powerSum_{k}[X(1-q)] = (1-q^k)\powerSum_{k}[X]$
\item $\powerSum_{k}[X/(1-q)] = \powerSum_{k}[X]/(1-q^k)$
\item $\powerSum_{k}[X+Y] = \powerSum_{k}[X]+\powerSum_{k}[Y]$
\item $\powerSum_{\lambda}[XY] = \powerSum_{\lambda}[X]\powerSum_{\lambda}[Y]$
\item $\schurS_{\lambda}[X+Y] = \sum_{\mu \subseteq \lambda} \schurS_{\mu}[X]\schurS_{\lambda/\mu}[Y]$
\item
$\schurS_\nu[XY] = \sum_{\lambda,\mu} g^{\nu}_{\lambda\mu} \schurS_\lambda(\xvec) \schurS_\mu(\yvec)$
where $g^{\nu}_{\lambda\mu}$ are the \hyperref[schurKroneckerCoefficients]{Kronecker coefficients}.
\end{itemize}
The last identity generalizes as follows:
\[
\schurS_\nu[f\cdot g] = \sum_{\lambda,\mu} g^{\nu}_{\lambda\mu} \schurS_\lambda[f] \schurS_\mu[g].
\]


We have not found a reference for the following useful relation, but it is
easy to prove.  It is used in \cite{AlexanderssonUhlin2020}.
\begin{proposition}[Alexandersson, 2019]
Let $f$ be a homogeneous symmetric function of degree $n$.
Then
\[
\powerSum_k[ \omega f ] = (-1)^{n(k+1)}\omega(\powerSum_k[ f ]).
\]
\end{proposition}


\subsection[plethysmOther]{Other identities}

\begin{proposition}
We have
\[
\completeH_k[\completeH_2] = \sum_{\mu : \text{ even }} \schurS_{\mu}
\]
where the sum ranges over all partitions of $2k$ into even parts.
This identity is due to \name{Dudley Littlewood}.
\end{proposition}

\name[Milo Bechtloff Weising]{Milo Bechtloff Weising} proves a wreath-product
generalization of Littlewood reciprocity \cite{Weising2025Littlewoodx}.
The result gives branching multiplicities for irreducible
$G^n\rtimes \symS_n$-representations inside highest-weight
$\mathrm{GL}_{nm}(\setC)$ representations, extending the classical
Littlewood rule for the embedded symmetric group.

The \defin{SXP rule}\label{sxpRule} gives a formula for the plethysm
\cite{Wildon2018}:
\[
\schurS_\lambda[\powerSum_r] = \sum_{\nuvec} \sign_r(\nuvec^\ast) c^{\lambda}_{\nuvec} \schurS_{\nuvec^\ast}.
\]
See also the \hyperref[plethysticMNRule]{plethystic Murnaghan--Nakayama rule}.
\name{William F. Doran} gives a tableau formula for
$\powerSum_\mu[\schurS_{(a)}]$ using roots of unity and a major-index-like
statistic on semistandard tableaux \cite{Doran1998}.
\name{Ira M. Gessel} and \name{Yan Zhuang} give plethystic formulas for
permutation enumeration \cite{GesselZhuang2020}.

\name[Florence Maas-Gariépy]{Florence Maas-Gariépy} and
\name{Étienne Tétreault} develop a tableau approach to plethysms
$\schurS_\nu[\completeH_\lambda]$ and $\schurS_\nu[\elementaryE_\lambda]$
\cite{MaasGariepyTetreault2022x}.  They use ribbon tableaux, lattice points
in polytopes, and a Kronecker map to analyze the case where $\nu$ is a
partition of $3$ and then extend to arbitrary $\lambda$.

\name{Álvaro Gutiérrez} and \name{Mercedes H. Rosas} prove partial symmetries
for large families of iterated plethysm coefficients
\cite{GutierrezRosas2023}.  Their flip symmetry generalizes the usual
$\omega$-involution symmetry and is preserved by certain plethystic
operations.
\name{Álvaro Gutiérrez}, \name{Rosa Orellana}, \name{Franco Saliola},
\name{Anne Schilling}, and \name{Mike Zabrocki} study bivariate generating
functions for plethysm coefficients with bounded output length
\cite{GutierrezOrellanaSaliolaSchillingZabrocki2025x}.  They prove
rationality, give an explicit $q$-Ehrhart-theoretic algorithm in the
two-row case, and derive linear recursions for $\mathrm{SL}_2$ plethysm
coefficients.
\name{Christine Bessenrodt}, \name{Chris Bowman}, and \name{Rowena Paget}
classify multiplicity-free plethysms of Schur functions
\cite{BessenrodtBowmanPaget2022}.
\name{Rowena Paget} and \name{Mark Wildon} study plethysms through
representations of $\mathrm{SL}_2(\setC)$ \cite{PagetWildon2021}.  They
classify several families of plethystic isomorphisms, including cases with
conjugate partitions, rectangles, and partitions with at most two rows or
columns.
\name{Igor Pak}, \name{Greta Panova}, and \name{Joshua P. Swanson} give a
positive combinatorial interpretation for plethysm coefficients
$\langle \schurS_\mu[\schurS_\nu],\schurS_\lambda\rangle$ when $\lambda$ has
at most two rows \cite{PakPanovaSwanson2025x}.  Their model counts explicit
marked trees and, in the case $\mu=(n)$, also interprets corresponding
rectangular Kronecker coefficients.



\begin{proposition}[See \cite[Appendix~B]{BentoNovaes2021}]
For an indeterminate $c$,
\[
\schurS_\lambda\left[ \frac{\xvec}{1- c \xvec} \right] = \sum_{\rho} c^{|\rho|-|\lambda|}
\det\left[  \binom{ \rho_j-j }{\lambda_i-i}   \right] \cdot \schurS_\rho(\xvec).
\]
The determinant of binomial coefficients has a combinatorial interpretation, see \cite{GesselViennot1985}.
\end{proposition}
Moreover, \name{Damir Yeliussizov} shows that the plethysm is in fact a
specialization of the \hyperref[grothendieckCanonical]{canonical stable
Grothendieck polynomial} \cite{Yeliussizov2017}:
\[
   \grothendieckStable^{(c,-c)}_\lambda(\xvec)
   = \schurS_\lambda\left[ \frac{\xvec}{1- c \xvec} \right].
\]


\section[plethysmComputation]{The s-perp trick}

In \cite{ColmenarejoOrellanaSaliolaSchillingZabrocki2022x}, the authors
describe in detail how to use the so-called
\defin{s-perp trick}\label{sPerpTrick}
in order to compute certain plethysm coefficients.
This trick can also be used in many other settings.  They give algorithms for
monomial and Schur expansions of symmetric functions, and in several special
cases these are faster than the corresponding SageMath routines.

First, define the following operator on a symmetric function $f$:
\[
 \schurS_\lambda^{\perp} f \coloneqq \sum_{\mu} \langle f , \schurS_\lambda \schurS_\mu \rangle \schurS_\mu.
\]

\begin{proposition}
Let $f$ and $g$ be two homogeneous symmetric functions of degree $d$.
Then
\[
 \schurS_r^{\perp} f = \schurS_r^{\perp} g \text{ for all $1\leq r \leq d$}
\]
implies that $f=g$.  The same holds if we replace
$\schurS_r = \completeH_r$ with $\schurS_{1^r} = \elementaryE_r$.
\end{proposition}


\section[plethysmProblems]{Open problems}

\subsection[plethysmWithSchur]{Schur plethysm}

\begin{problem}[See \cite{LoehrRemmel2010} for more info]
Let $\lambda \vdash n$ and $\mu \vdash m$.  Find a formula for the
\defin{Schur plethysm coefficients}\label{schurPlethysmCoefficient} in
\[
\schurS_\lambda[\schurS_\mu] = \sum_{\nu \vdash mn} p^{\nu}_{\lambda \mu} \schurS_{\nu}.
\]
This is a major open problem in the theory of symmetric functions and representation theory of classical groups.
\end{problem}

\begin{example}
For example,
\[
\schurS_{211}[\schurS_{11}] =
\schurS_{3 3 2} + \schurS_{3 2 2 1} + \schurS_{4 2 1 1} +
 \schurS_{2 2 2 1 1} + \schurS_{3 2 1 1 1} + \schurS_{3 1 1 1 1 1}.
\]
\end{example}


The Schur plethysm coefficients \emph{stabilize} under some operations.
In \cite{PagetWildon2025} Paget and Wildon prove a general theorem,
which specializes to all known stability results regarding plethysm coefficients.
For example, the main stability result of \cite{LawOkitani2023} is proved.

\subsection[plethysmFoulkesConjecture]{Foulkes conjecture}

\defin{Foulkes conjecture}\label{foulkesConjecture} \cite{Foulkes1950}
states that $\completeH_b[\completeH_a]-\completeH_a[\completeH_b]$ is
\hyperref[schurPositivity]{Schur-positive} whenever $a \leq b$.
See the \hyperref[cycleIndexPolynomial]{cycle index polynomial} page for more information.
\name{{\'E}tienne T{\'e}treault} introduces a relation on integer partitions
motivated by Schur positivity of differences of plethysms
\cite{Tetreault2020PartitionOrder}.  The restriction to one-part partitions is
equivalent to the classical Foulkes conjecture, and the paper proves several
properties via explicit inclusions of modules.

\name{Hai Zhu} studies plethysm through orbit harmonics of loci of unordered
set partitions \cite{Zhu2026x}.  Applying orbit harmonics to
$\Pi_{(2^a)}$ and $\Pi_{(a^2)}$ gives graded $\symS_{2a}$-modules whose
graded characters refine the Schur expansions of $\completeH_a[\completeH_2]$
and $\completeH_2[\completeH_a]$.  The paper also formulates a
Foulkes-type conjecture from the pair $\Pi_{(b^a)}$, $\Pi_{(a^b)}$ and proves
the case $b=2$.

\begin{example*}[A small orbit-harmonics refinement]
For $a=3$, Zhu's graded Frobenius formulas
\cite[Thms.~3.3 and 3.5]{Zhu2026x} give
\[
  \operatorname{grFrob}(R(\Pi_{(2^3)});q)
  = \schurS_{6}+q\schurS_{42}+q^2\schurS_{222}
\]
and
\[
  \operatorname{grFrob}(R(\Pi_{(3^2)});q)
  = \schurS_{6}+q\schurS_{42}.
\]
After forgetting the grading by setting $q=1$, these become the plethysms
\[
  \completeH_3[\completeH_2]
  = \schurS_{6}+\schurS_{42}+\schurS_{222}
  \quad\text{and}\quad
  \completeH_2[\completeH_3]
  = \schurS_{6}+\schurS_{42}.
\]
Thus in this first nontrivial case of Foulkes' conjecture, the difference is
the single Schur-positive term
\[
  \completeH_3[\completeH_2]-\completeH_2[\completeH_3]
  = \schurS_{222}.
\]
\end{example*}
