Schubert polynomials were introduced by \name{Alain Lascoux} and
\name[M.-P. Schützenberger]{Marcel-Paul Schützenberger}
in \cite{LascouxSchutzenberger1982}.
The Schubert polynomials generalize the \hyperref[schurS]{Schur polynomials} and
represent Schubert cycles in \hyperref[flagVarieties]{flag varieties}; this is
one of the polynomial models behind \hyperref[schubertCalculus]{Schubert calculus}.
The Schubert polynomials are generalized by the \hyperref[grothendieck]{Grothendieck polynomials},
which play the same role in \hyperref[kTheory]{K-theory}.
\name[Kursat Aker]{K{\"u}r{\c{s}}at Aker} and
\name[Nesrin Tutas]{Nesrin Tuta{\c{s}}} define rational Schubert, rational
Grothendieck, and rational key polynomials \cite{AkerTutas2015}.  Their
motivation is to view \name{Alexander Molev}'s dual Schur functions through
Lascoux-style divided-difference constructions.
\name[Neil J. Y. Fan]{Neil J.~Y.~Fan} and \name{Peter L. Guo} give an elementary
divided-difference proof of the Buch--Rimanyi specialization formula for
\hyperref[grothendieckDouble]{double Grothendieck polynomials}
\cite{FanGuo2018x}.
\name{Zachary Hamaker}, \name{Oliver Pechenik}, \name{David E. Speyer}, and
\name{Anna Weigandt} study a differential operator on Schubert polynomials
\cite{HamakerPechenikSpeyerWeigandt2020}.  Their application gives a short
proof of an identity of \name{Ian G. Macdonald} and proves a determinant
conjecture of \name{Richard P. Stanley}.
\name{Colleen Robichaux}, \name{Harshit Yadav}, and \name{Alexander Yong}
survey edge-labeled Young tableaux as models for equivariant Schubert
calculus of Grassmannians \cite{RobichauxYadavYong2022}.  They also report on
a shifted analogue connected to isotropic Grassmannians.

The Schubert polynomials are indexed by permutations.  For fixed $n$, the set
\[
\{ \schubert_\omega(x_1,\dotsc,x_n) \}_{\omega \in \symS_n}
\]
is a basis for the quotient $\setZ[x_1,\dotsc,x_n]/I_n$
where $I_n = \langle \elementaryE_1, \elementaryE_2,\dotsc, \elementaryE_n \rangle$,
that is, the ideal generated by nonconstant symmetric functions.
The Schubert polynomials $\schubert_\omega$ with $\omega \in \symS_\infty$
form an integral basis for $\setZ[x_1,x_2,\dotsc]$.


\subsection[schubertOperatorformula]{Operator definition}

Consider the ring $\setR[x_1,\dotsc,x_n]$ and let $s_i$ act by permuting
$x_i$ and $x_{i+1}$.
Define
\[
  \partial_i(f)=\frac{f-s_i(f)}{x_i-x_{i+1}},
  \qquad 1 \leq i \leq n-1.
\]
These \defin{divided difference operators}\label{dividedDifferenceOperators}
satisfy the
\hyperref[braidRelations]{braid relations},
and we let $\partial_\omega = \partial_{a_1} \partial_{a_2} \dotsm \partial_{a_k}$
where $(a_1,a_2,\dotsc,a_k)$ is a \hyperref[reducedWord]{reduced word} for $\omega$.

The \defin{Schubert polynomials} are then defined as
\begin{equation*}
 \schubert_\omega(x_1,\dotsc,x_n)
 \coloneqq
 \partial_{\omega^{-1}\omega_0}(x_1^{n-1}x_2^{n-2}\dotsm x_{n-1}).
\end{equation*}


\subsection[schubertOrthodontia]{Orthodontia formula}

\name{Peter Magyar} gives a Demazure-operator formula for Schubert
polynomials using Bott--Samelson varieties
\cite{Magyar1998SchubertBottSamelson}.  In later combinatorial work, this
is often phrased as Magyar's \defin{orthodontia formula}: one applies an
algorithm to the diagram and records a word of Demazure operators whose
product recovers the Schubert polynomial.  The
\hyperref[grothendieckOrthodontia]{Grothendieck orthodontia formula} is the
$K$-theoretic analogue.


\subsection[schubertReducedwordFormula]{Reduced word formula}
Let $RW(\omega)$ be the set of reduced words of $\omega$.

We say that a $k$-tuple $\alpha_1,\dotsc,\alpha_k$ is
\defin{$a$-compatible}\label{aCompatibleSequence}
if
\begin{itemize}
\item
$\alpha_1 \leq \alpha_2 \leq \dotsb \leq \alpha_k$
\item
$\alpha_j \leq a_j$ for $1 \leq j \leq k$
\item
$\alpha_j \lt \alpha_{j+1}$ whenever $a_j \lt a_{j+1}$.
\end{itemize}
Let $C(a)$ denote the set of $a$-compatible sequences.
Then
\begin{equation*}
 \schubert_\omega(x_1,\dotsc,x_n) = \sum_{a \in RW(\omega)} \sum_{\alpha \in C(a)}
 x^\alpha .
\end{equation*}



\subsection[schubertPipeDream]{Pipe dream formula}

The set of reduced words and compatible sequences can be
described in a more combinatorial manner using rc-graphs.
Suppose $(\alpha_1,\dotsc,\alpha_k)$ is an $a$-compatible sequence, for $a\in RW(\omega)$.
Let
\[
D(a,\alpha) = \{ (\alpha_j, a_j - \alpha_j + 1) \text{ for } j=1,\dotsc,k \}.
\]
This set of points is illustrated in an \defin{rc-graph}\label{rcGraph}, also
known as a \defin{pipe dream}\label{pipeDream}.

\begin{example}
% Related Rust: rust/sym-poly/multipoly/examples/schubert_site_example.rs
Let $\omega = 314652$. Then $a = 521345$ is in $RW(\omega)$,
and $\alpha = 111235$ is $a$-compatible.
We mark the corresponding entries in $D(a,\alpha)$ with a cross, $+$, in the rc-graph:

\begin{figure}
\svgimg[width=0.35\textwidth]{svg-images/pipe-dream-schubert-314652.svg}{
A pipe dream for the permutation 314652, with rows and columns indexed from
the upper-left corner.}
\end{figure}

Here
\[
  D(a,\alpha)=\{(1,5),(1,2),(1,1),(2,2),(3,2),(5,1)\},
\]
so this pipe dream contributes the monomial
\[
  \xvec_D=x_1^3x_2x_3x_5
\]
to $\schubert_{314652}$.
\end{example}

Let $RC(\omega)$ denote the set of all such diagrams.
Let $\xvec_D = \prod_{(i,j) \in D} x_i$. Then the Schubert polynomial is given as
\begin{equation*}
 \schubert_\omega(x_1,\dotsc,x_n) = \sum_{D \in RC(\omega)} \xvec_D.
\end{equation*}

See also \cite{FominKirillov1996YangBaxter}.

\name{Allen Knutson} and \name{Paul Zinn-Justin} introduce
\defin{generic pipe dreams}\label{genericPipeDream}, a tile model whose leading
forms recover the
classical pipe dream and bumpless pipe dream formulas for double Schubert
polynomials \cite{KnutsonZinnJustin2024x}.  Their generic pipe dream
polynomials compute equivariant classes of lower-upper varieties and also
appear in formulas for Segre--Schwartz--MacPherson classes.
\name{Ilani Axelrod-Freed} introduces inversion tableaux, a tableau model for
Schubert polynomials and Stanley symmetric functions
\cite{AxelrodFreed2025x}.  The model modifies Edelman--Greene balanced
staircase tableaux, specializes to semistandard Young tableaux in the
Grassmannian case, and is compatible with generalized chute moves.

\cite{LamLeeShimozono2022}.  This includes back stable Grothendieck

\subsection[schubertRepresentationTheory]{Representation theory}

This subsection follows the introduction in \cite{FanGuo2019}.

Let $D=(D_1,\dotsc,D_n)$ be an $n$-tuple of subsets of $[n]$.
Then $D$ encodes a subset of $[n]\times [n]$, where $D_j$ is considered as a
set of row indices (seen as a strictly increasing list).
For example, the following diagram encodes $(23,14,\emptyset,124)$
\begin{figure}
\begin{ytableau}
\; & *(lightblue) & \; & *(lightblue) \; \\
*(lightblue) &  & \; & *(lightblue) \; \\
*(lightblue) & \;  & \; &  \; \\
\; & *(lightblue) & \; &  *(lightblue) \; 
\end{ytableau}
\end{figure}
We define a partial order on such diagrams, where $C \leq D$ if for every $j \in [n]$,
we have $|C_j| = |D_j|$ and $C_{jk} \leq D_{jk}$ for all $k$.

Let $B \subset GL_n(\setC)$ be the set of invertible upper-triangular
$n\times n$ matrices.
Let $Y$ denote the upper-triangular matrix with indeterminate entries $\{y_{ij}\}_{i\leq j}$,
and let $\setC[Y]$ be the polynomial ring with these indeterminates.
A matrix $M \in GL_n(\setC)$ acts on $f \in \setC[Y]$ on the right by
$f \cdot M \coloneqq f(M^{-1} Y)$.

Given a diagram $D$, the \defin{flagged Weyl module}\label{flaggedWeylModule}
$\mathcal{M}_D$ is a $B$-module, defined as
\[
  \mathcal{M}_D
  \coloneqq
  \mathrm{Span}
  \left\{
    \prod_{j=1}^n \det( Y_{D_j}^{C_j} ) : C \leq D
  \right\}
  \subseteq \setC[Y],
\]
where $Y_{D_j}^{C_j}$ denotes the submatrix of $Y$ with row indices from
$C_j$ and column indices from $D_j$.

The character, $\mathrm{char}(\mathcal{M}_D)(x_1,\dotsc,x_n)$, is defined as
the trace of the map $X : \mathcal{M}_D \to \mathcal{M}_D$,
where $X = (x_1,\dotsc,x_n)$ is a diagonal matrix acting on $\setC[Y]$.
Note that for $C \leq D$,
\[
\left( \prod_{j=1}^n \det( Y_{D_j}^{C_j} ) \right) \cdot X =
 \prod_{j=1}^n  \prod_{i \in C_j} x_i^{-1} \det( Y_{D_j}^{C_j} )
\]
so all polynomials $\det( Y_{D_j}^{C_j} )$ with $C \leq D$ are eigenvectors,
and they span $\mathcal{M}_D$.
It was shown in \cite{KraskiewiczPragacz1987,KraskiewiczPragacz2004}
that the Schubert polynomial indexed by $\omega$ is related to the character
of $\mathcal{M}_{D(\omega)}$, via
\[
\schubert_\omega(x_1,\dotsc,x_n) = \mathrm{char}(\mathcal{M}_{D(\omega)})(x^{-1}_1,\dotsc,x^{-1}_n).
\]

\name{Zhuowei Lin}, \name[Simon C. Y. Peng]{Simon C.~Y.~Peng}, and
\name[Sophie C. C. Sun]{Sophie C.~C.~Sun} prove the
\name{Karola Mészáros}--\name[Avery St. Dizier]{Avery St.~Dizier}--
\name{Shijie Tanjaya} upper-bound conjecture for dual flagged Weyl characters
\cite{LinPengSun2023x}.  The criterion says that the natural upper bound is
attained exactly for diagrams avoiding a specified forbidden subdiagram.  Rothe
diagrams and skyline diagrams give the Schubert and key-polynomial special
cases.

\name{Yibo Gao}, \name{Reuven Hodges}, and \name{Alexander Yong} classify
Levi-spherical Schubert varieties \cite{GaoHodgesYong2023}.  Their proof uses
type $A$ Demazure characters, i.e. \hyperref[key]{key polynomials}, and gives
a formulation in terms of standard Coxeter elements.

\name{Peter L. Guo}, \name{Zhuowei Lin}, and \name[Simon C. Y. Peng]{Simon C.~Y.~Peng}
prove the zero-one criterion for dual characters of flagged Weyl modules
\cite{GuoLinPeng2025}.  A diagram $D$ has zero-one dual character if and only
if it avoids the twelve multiplicitous configurations of
\cite[Fig.~1.1]{GuoLinPeng2025}.  Since Schubert and key polynomials are
special cases of dual flagged Weyl characters, this gives a unified proof of
the known zero-one criteria for Schubert and key polynomials.
For Schubert polynomials, \name{Alex Fink}, \name{Karola Mészáros}, and
\name[Avery St. Dizier]{Avery St.~Dizier} prove that the zero-one property is
closed under \hyperref[permutationPattern]{permutation pattern} containment
and characterize it by avoidance of twelve patterns
\cite{FinkMeszarosDizier2019x}.  \name{Anna Weigandt} gives a lower bound for
the sum of coefficients in terms of $132$-pattern containment
\cite{Weigandt2018}, while Fan and Guo characterize when Schubert polynomials
attain the upper bound coming from the Rothe diagram by avoidance of the two
patterns $1432$ and $1423$ \cite{FanGuo2019}.

\name{David Anderson} gives a representation-theoretic interpretation of the
\name{Philippe Nadeau}--\name{Hunter Spink}--\name{Vasu Tewari} recursion for
Schubert polynomials \cite{Anderson2024SchubertModulesx}.  Using filtrations of
Kraskiewicz--Pragacz Schubert modules, Anderson proves that the recursion
determines the characters of these modules, and extends the same mechanism to
flagged Schur modules coming from transparent and translucent diagrams.


\subsection[schubertGTFormula]{Gelfand--Tsetlin polytopes}

In \cite{LiuMeszarosDizier2019}, the authors express the Schubert polynomials
as a projection of a Minkowski sum of \hyperref[gtpolytopes]{Gelfand--Tsetlin polytopes}.
This generalizes the way Schur polynomials are expressible as
an integer point transform of a single Gelfand--Tsetlin polytope.

\name{Federico Castillo}, \name{Yairon Cid-Ruiz}, \name{Fatemeh Mohammadi},
and \name{Jonathan Montaño} prove that double Schubert polynomials have
saturated \hyperref[newtonPolytope]{Newton polytopes}
\cite{CastilloCidRuizMohammadiMontano2023}.
Thus every lattice point in the Newton polytope occurs as an exponent vector.

See also a similar result in \cite{FinkMeszarosDizier2018},
using the integer point transforms of
\hyperref[generalizedPermutohedron]{generalized permutohedra}.

Top-degree components of Lascoux polynomials provide another bridge to
Schubert polynomials.  In \cite{Yu2025}, \name{Tianyi Yu} shows that
top Lascoux polynomials are reverse-complement transforms of Schubert
polynomials and that their structure constants coincide with Schubert
structure constants.

\name{Sylvester W. Zhang} relates back stable Schubert calculus to the
Heisenberg algebra \cite{Zhang2024x}.  The Hilbert space with basis indexed by
infinite permutations and the ring of back symmetric functions both carry
faithful Heisenberg actions, and the map sending an infinite permutation to
its back stable Schubert polynomial is an isomorphism of representations.
The same framework interprets the pipe dream model as Hamiltonian time
evolution of two-dimensional fermions.



\subsection[schubertBlockDecomposition]{Block decomposition formula}

If $\sigma \in \symS_j$ and $\tau \in \symS_{k}$,
we can form the \hyperref[directSumPermutations]{skew sum}
$\sigma \times \tau \in \symS_{j+k}$.
We then have the factorization (see \cite{BilleyJockuschStanley1993})
\[
 \schubert_{\sigma \times \tau} = (x_1 \dotsm x_j)^k \schubert_{\sigma} \; \uparrow^j \! \schubert_{\tau}
\]
where $\uparrow^j x_i \coloneqq x_{i+j}$ is the operator which
increases the variable index by $j$ on all variables.


\subsection[grassmannianPermutation]{Grassmannian permutations}

A permutation $\omega \in \symS_n$ is called \defin{Grassmannian} (of descent $k$)
if $\omega_i \lt \omega_{i+1}$ for all $i\neq k$.
All Grassmannian permutations are \hyperref[vexillary]{vexillary}, meaning that
they avoid the permutation pattern $2143$.

A Grassmannian permutation determines a partition:
\[
\lambda(\omega) = (\omega_k - k, \omega_{k-1} - k + 1, \dotsc, \omega_1 - 1 ).
\]
The same partitions index Schubert cells in the \hyperref[grassmannian]{Grassmannian}
$\mathrm{Gr}(k,n)$.  Fix a complete flag
$0=F_0\subset F_1\subset \dotsb \subset F_n=\setC^n$.
For $\lambda$ contained in the $k\times(n-k)$ rectangle, the Schubert variety
$\Omega_\lambda(F_\bullet)$ is cut out by the rank conditions
\[
  \dim(V\cap F_{n-k+i-\lambda_i}) \geq i,\qquad 1\leq i\leq k.
\]
Its open cell $\Omega^\circ_\lambda(F_\bullet)$ has complex codimension
$|\lambda|$, and these cells form the standard cell decomposition of the
Grassmannian.  The corresponding Schubert class is represented by the Schur
polynomial $\schurS_\lambda$ in the cohomology ring of the Grassmannian; see
\cite[Ch.~9]{Fulton1997}.
Thus
\begin{equation*}
\schurS_{\lambda(\omega)}(x_1,\dotsc,x_k) = \schubert_{\omega}(\xvec).
\end{equation*}

Furthermore, the \hyperref[stanleySym]{Stanley symmetric functions}
are obtained as a stable limit of the Schubert polynomial.


\subsection[schubertMurnaghanNakayama]{Murnaghan--Nakayama rule}

In \cite{MorrisonSottile2018}, \name{Kevin N. Morrison} and \name{Frank Sottile}
prove a \hyperref[murnaghanNakayamaRule]{Murnaghan--Nakayama rule} for
multiplying a Schubert polynomial by
$\powerSum_r(x_1,\dotsc,x_k)$.
The expansion is indexed by certain cycles in the \hyperref[bruhatOrder]{$k$-Bruhat order},
with signs determined by a descent statistic on the cycle.
In the Grassmannian case this recovers the classical border-strip rule for
Schur functions.

\subsection[schubertMonksRule]{Monk's rule}

The geometric version of this result was given by \name{David Monk} in \cite{Monk1959}.

Let $s_r$ be a simple transposition. Then
\[
\schubert_{s_r}(\xvec) \schubert_\omega(\xvec) = \sum \schubert_{\omega t_{ij}}(\xvec)
\]
where the sum is over all transpositions $t_{ij}$ such that $i\leq r \lt j$ and
$\length(\omega t_{ij}) = \length(\omega)+1$.
Notice that
\[
\schubert_{s_r}(\xvec) = x_1+x_2+\dotsb + x_r.
\]

A combinatorial proof is given in \cite{CoskunTaskin2018}, and a bijective proof using bumpless pipe
dreams can be found in \cite{Huang2020x}.
Huang also proves Monk's rule for the double Schubert polynomials with this model.

\subsection[schubertPieriRule]{Pieri rule}

\name{Frank Sottile} provides the following Pieri-type formulas for Schubert
polynomials \cite{Sottile1996}.
Let $\omega \in \symS_n$, then
\[
\completeH_{m}(x_1,\dotsc,x_k) \schubert_\omega(\xvec) = \sum_{\omega'} \schubert_{\omega'}(\xvec)
\]
where the sum runs over all $\omega' = \omega t_{a_1b_1}\dotsm t_{a_mb_m}$ such that $a_i \leq k \leq b_i$,
$\length(\omega t_{a_1b_1}\dotsm t_{a_ib_i}) = \length(\omega)+i$ and all $b_i$ distinct.

Similarly,
\[
\elementaryE_{m}(x_1,\dotsc,x_k) \schubert_\omega(\xvec) = \sum_{\omega'} \schubert_{\omega'}(\xvec)
\]
with the same sum as above, but now the $a_i$ are distinct.


\subsection[schubertSottileRule]{Sottile rule}

Sottile also gives a rule for multiplying $\schubert_\omega(\xvec)$ by a Schur
polynomial in the first $k$ variables when the Schur shape is a hook
\cite{Sottile1996}.
This sits between Monk's rule and the Schubert
\hyperref[schubertLittlewoodRichardson]{Littlewood--Richardson problem}: it is
positive and is controlled by chains in the \hyperref[bruhatOrder]{$k$-Bruhat order}.


\subsection[schubertCauchyFormula]{Cauchy identity}

Setting $w=\omega_0$ in the
\hyperref[schubertDoubleGiambelliFormula]{Giambelli formula} for double
Schubert polynomials gives the following
\defin{Cauchy identity} for Schubert polynomials:
\[
\prod_{i+j \leq n} (x_i - y_j) = \sum_{\omega} \schubert_{\omega}(\xvec) \schubert_{\omega_0 \omega}(-\yvec).
\]


\subsection[schubertLittlewoodRichardson]{Littlewood--Richardson rule}


\begin{problem}
Give a combinatorial interpretation of the coefficients in the product
\[
\schubert_u(\xvec) \schubert_v(\xvec) = \sum_{w} c^w_{uv} \schubert_w(\xvec).
\]
\end{problem}
Because of the geometric interpretation of Schubert polynomials,
it is known that the coefficients $c^w_{uv}$ are nonnegative integers.

Note that the coefficients $c^w_{uv}$ generalize the classical Littlewood--Richardson
coefficients in \hyperref[schurLittlewoodRichardson]{the Littlewood--Richardson rule for Schur polynomials}.
Finding a positive combinatorial rule is a major open problem in
\hyperref[schubertCalculus]{Schubert calculus} and algebraic combinatorics.
\name{Nantel Bergeron} and \name{Frank Sottile} relate these structure
constants to the \hyperref[bruhatOrder]{Bruhat order} and the geometry of
\hyperref[flagVarieties]{flag manifolds}
\cite{BergeronSottile1998}.  They prove identities among Schubert structure
constants, derive formulas for several cases, obtain chain-enumeration results
in \hyperref[bruhatOrder]{Bruhat order}, and introduce a graded partial order
on the symmetric group containing Young's lattice.
\name{Igor Pak} and \name{Colleen Robichaux} disprove the proposed saturation
property for Schubert coefficients \cite{PakRobichaux2026x}.  They show that
the failure occurs for a large family of instances, also refute saturation
under bit scaling, and discuss consequences for computational complexity.

This problem is closely related to
\defin{the Fomin--Kirillov conjecture}\label{fominKirillovConjecture},
see \cite[Conjecture 8.1 and Problem 8.3]{FominKirillov1999}.
It concerns the evaluation of Schubert polynomials at Dunkl elements.


There are many special cases proved for this problem.  One such case,
concerning products of two permutations with separated descents, is treated
in \cite{Huang2022}.

\name{Oliver Pechenik} and \name{Anna Weigandt} introduce backstable clans and
prove an inverse Grassmannian Littlewood--Richardson rule
\cite{PechenikWeigandt2024}.  Their work also gives new linear relations among
Schubert structure constants, extending the range of products accessible by
positive combinatorial formulas.

\name{Matthew Samuel} maintains \texttt{schubmult}, a Python package for
computing products of Schubert polynomials.


\begin{problem}
Can one give an efficient algorithm to decide if $c^w_{uv} = 0$ or not?
\end{problem}
This problem is discussed in \cite{PakRobichaux2025xI}, where it is shown
that, assuming GRH, this problem lies in $AM \cap coAM$,
where $AM$ denotes the Arthur--Merlin probabilistic proof class.



\subsection[schubertKeyExpansion]{Expansion in key polynomials}

There are several proofs of the fact that Schubert polynomials expand positively into \hyperref[key]{key polynomials},
the first proof appearing in \cite{LascouxSchutzenberger1990}.

One recent proof is by using crystal bases, see \name{Sami Assaf} and \name{Anne Schilling} \cite{AssafSchilling2018}.
A crystal structure on RC-graphs where the key expansion is obtained is given
by \name{Sarah Gold}, \name{Elizabeth Milićević}, and \name{Yuxuan Sun} in \cite{GoldMilicevicSun2024x}.

\subsection[schubertAsSumOfElementary]{Sum of elementary}

Let $\omega \in \symS_{n+1}$, and let $\elementaryE_{j}(m) = \elementaryE_{j}(x_1,\dotsc,x_m)$
denote the elementary symmetric functions of degree $j$ in $m$ variables.
Then there is a unique way to express $\schubert_\omega(\xvec)$ as
\[
\schubert_\omega(\xvec) = \sum a_{k_1 \dotsc k_n} \elementaryE_{k_1}(1) \elementaryE_{k_2}(2) \dotsm \elementaryE_{k_n}(n)
\]
where the sum ranges over integers such that $0 \leq k_i \leq i$ and $k_1+\dotsb + k_n = \length(\omega)$.
This expansion is the starting point for defining the quantum Schubert polynomials.
The coefficients $a_{k_1 \dotsc k_n}$ are in general not always positive.
These coefficients are studied in \cite{Winkel1998} but many questions are left unanswered.

\name{Dora Woodruff} characterizes the permutations $w$ for which the
Schubert polynomial $\schubert_w$ is a single standard elementary monomial
\cite{Woodruff2025x}.  The characterization is by pattern avoidance and also
has an analogue for complete homogeneous monomials.

\name{Oma Makhija} studies when a Schubert polynomial factors as a product of
elementary symmetric polynomials \cite{Makhija2025x}.  The paper conjectures a
pattern-avoidance characterization, proves one direction, and gives
obstructions coming from rectangular arrays of crosses in bottom pipe dreams.



\begin{problem}[See \cite{MerzonSmirnov2015}]
For each $n$, find a permutation $w\in\symS_n$
which maximizes $\schubert_w(1,1,\dotsc,1)$ and find the maximal value.
\end{problem}
The sequence is given as \oeis{A284661}.
See also \cite{Stanley2017SchubertShenanigans}.

\name{Philippe Nadeau} and \name{Vasu Tewari} relate the permutahedral
variety, mixed Eulerian numbers, and principal specializations of Schubert
polynomials \cite{NadeauTewari2021}.
\name{Hugh Dennin} proves pattern lower bounds for principal specializations
of Schubert polynomials in the $1243$-avoiding case and extends the result to
$\beta$-Grothendieck polynomials for vexillary $1243$-avoiding permutations
\cite{Dennin2025}.  The proof gives bijective interpretations of the
coefficients appearing in these bounds.

\name{Karola Mészáros} and \name{Arthur Tanjaya} prove nonnegativity of an
inclusion--exclusion expression for Schubert polynomials of permutations
avoiding $1432$ and $1423$ \cite{MeszarosTanjaya2021x}.  This gives a partial
answer to a conjecture of \name{Yibo Gao} on principal specializations of
Schubert polynomials.
\name{Gleb Nenashev} studies differential operators on Schur and Schubert
polynomials \cite{Nenashev2020x}.
\name{Hugh Dennin} gives a pipe-dream proof of the Gaetz--Tung positivity
result for a dual differential operator acting on padded Schubert polynomials
\cite{Dennin2025x}.
\name{Peter L. Guo} and \name{Zhuowei Lin} give lower bounds for the number
of support monomials of a Schubert polynomial, equivalently lattice points in
its \hyperref[newtonPolytope]{Newton polytope}, in terms of
\hyperref[permutationPattern]{permutation-pattern occurrences}
\cite{GuoLin2024x}.  Their framework is the more general setting of dual
characters of flagged Weyl modules.



\subsection[schubertComplexity]{Complexity}

Deciding whether a monomial coefficient, or a Schubert structure constant
coefficient $c^w_{uv}$, is nonzero can be done in probabilistic polynomial
time; see \cite{AdveRobichauxYong2021x,PakRobichaux2025xII}.
\name[Avery St. Dizier]{Avery St.~Dizier} and \name{Alexander Yong} connect
\hyperref[generalizedPermutohedron]{generalized permutahedra} with
\hyperref[schubertCalculus]{Schubert calculus}
\cite{DizierYong2020SchubertPermutahedra}.  They obtain sufficient vanishing
criteria for Schubert intersection numbers, using Schubitopes as Newton
polytopes of Schubert polynomials, and give a polynomial-time tableau test.
\name{Igor Pak} and \name{Colleen Robichaux} prove that the vanishing problem
for Schubert coefficients lies in $\mathrm{coAM}$ assuming the generalized
Riemann hypothesis \cite{PakRobichaux2024Vanishing}.  They also formulate a
conditional positive rule for deciding Schubert coefficient positivity
\cite{PakRobichaux2024Positivity}.
\name{Igor Pak} and \name{Colleen Robichaux} also give a signed puzzle rule
for Schubert coefficients based on Knutson's recurrence
\cite{PakRobichaux2025SignedPuzzles}.  They use it to prove polynomiality
for sums of Schubert coefficients with bounded number of inversions.

twisted variants, see the companion page on
