\subsection*{Complex Definitions}

\begin{itemize}
    \item \defin{Richardson permutation}.
    With parameters $(k_1,k_2,\dotsc,k_m)$, it is the direct sum of the longest permutations in $\symS_{k_i}$:
    \[
        \begin{pmatrix}
        1 & 2 & \dotsc & k_1 & k_1+1 & \dots & k_1 + k_2 & \dots \\
        k_1 & k_1-1 & \dots & 1 & k_1+k_2 & \dots & k_1+1 & \dots \\
        \end{pmatrix}
    \]

    \item \defin{Wachs permutations} \oeis{A359039}.
    Defined as $W_n \coloneqq \{ \sigma \in \symS_n : |\sigma^{-1}(i)-\sigma^{-1}(i^*)|\leq 1 \}$, where:
    \[
     i^* \coloneqq \begin{cases}
     i-1 &\text{if $i$ is even}  \\
     i+1 &\text{if $i$ is odd and $i+1 \leq n$}  \\
     n & \text{otherwise.}
     \end{cases}
    \]
\end{itemize}
