\metatitle{Real-rooted and stable polynomials}

\metadescription{Polynomials with real roots, and their multivariate analogues. Also, Lorentzian polynomials.}


\section[polynomialRootIntro]{Introduction}

For an enumerative family of polynomials, natural questions include
\emph{unimodality}, \emph{real-rootedness}, \emph{stability}, and
\emph{interlacing roots}.

A useful general reference is \cite{Branden2015}, together with other works
of \name{Petter Brändén}.  See also the IPAC seminar lecture on Lorentzian
polynomials \cite{IPAC2022LorentzianLecture}.

The following pages cover more specialized topics:
\begin{itemize}
\item \hyperref[interlacingPolynomials]{Interlacing polynomials}:
  interlacing roots, operators preserving real-rootedness, Sturm sequences.
\item \hyperref[polyaFrequency]{Pólya frequency sequences and total positivity}:
  Toeplitz matrices, total non-negativity, Veronese sections, and path matrices.
\item \hyperref[realRootedWords]{Permutation statistics}:
  Eulerian polynomials, peaks, derangements, and other permutation statistics.
\item \hyperref[realRootedCatalan]{Catalan families}:
  Narayana polynomials, ballot polynomials, Motzkin polynomials.
\item \hyperref[realRootedTableaux]{Tableaux and polytopes}:
  descent polynomials of standard Young tableaux, Ehrhart polynomials.
\item \hyperref[realRootedMatching]{Graph polynomials}:
  matching polynomials, independence polynomials, rook polynomials.
\item \hyperref[stablePolynomial]{Stable polynomials}:
  multivariate stability and the half-plane property.
\item \hyperref[lorentzianPolynomial]{Lorentzian polynomials}:
  ultralog-concavity and the symmetric decomposition of $h^*$-polynomials.
\item \hyperref[gammaPositivity]{Gamma-positive polynomials}:
  palindromic polynomials, the Foata--Strehl action, and Gal's conjecture.
\end{itemize}

\section[polynomialUnimodality]{Unimodality, log-concavity and real roots}

Let $P = a_0 + a_1t+\dotsb + a_n t^n$ be a polynomial.
It is called \defin{unimodal}\label{unimodalPolynomial} if there is some $j$
such that
\[
a_0 \leq a_1 \leq \dotsb \leq a_j \geq a_{j+1}
\geq a_{j+2} \geq \dotsb \geq a_n.
\]

Suppose now all coefficients are positive. We say that \(P\) is
\defin{log-concave}\label{logConcavePolynomial} if
\[
  a_{j-1}a_{j+1} \leq a_j^2
\]
for all relevant $j$. This property implies unimodality.

We get an even stronger property if the
\defin{Newton inequalities}\label{newtonInequalities} are satisfied:
\[
  \frac{a_{j-1}}{ \binom{n}{j-1} } \frac{a_{j+1}}{ \binom{n}{j+1} } \leq \frac{a_{j}^2}{ \binom{n}{j}^2 }.
\]

Finally, if $P$ has only real roots, it is called
\defin{real-rooted}\label{realRootedPolynomial}.
Every real-rooted polynomial satisfies the Newton inequalities, see e.g. \cite{Branden2015}.


In \cite{Wang2002}, a conjecture by \name{Rodica Simion} is proved:
a sequence of lattice path counts is log-concave.
See also \cite{Sagan1987unimodal} for a short proof of unimodality in this case.

In \cite{AhmiaBelbachir2018} it is shown that if $f(t)$
has non-negative, non-decreasing coefficients, then the polynomial
$f(1+x+x^2+\dotsb+x^\ell)$ is unimodal,
for any $\ell \geq 1$.
\name{Ethan Y. H. Li}, \name{Grace M. X. Li}, \name{Arthur L. B. Yang}, and
\name{Zhong-Xue Zhang} use chromatic symmetric functions to prove
log-concavity results for independence polynomials \cite{LiLiYangZhang2025x}.
Their applications include spiders and pineapple graphs.
\name{Jinting Liang} and \name{Bruce E. Sagan} give a broad distributive
lattice method for proving log-concavity and log-convexity
\cite{LiangSagan2024x}.  Their Order Ideal Lemma applies to lattice paths,
Young's lattice intervals, order polynomials, Schur and Schur \(Q\)-function
specializations, Lucas sequences, descent and peak polynomials, pattern
avoidance, set partitions, and noncrossing partitions.



\subsection[kurtzTheorem]{Kurtz theorem}

\begin{theorem}[Kurtz sufficiency theorem, \cite{Hutchinson1923,Kurtz1992}]
\name{David Kurtz} proved that a polynomial
$a_n t^n + a_{n-1}t^{n-1} + \dotsb + a_1 t + a_0$
with positive coefficients (and degree at least two) that satisfies
\[
 a_i^2 \gt 4a_{i-1}a_{i+1} \text{ for all } i,
\]
is real-rooted.
\end{theorem}
This result was proved earlier in the context of entire functions, see \cite{Hutchinson1923}.


\section[realRootedTransformations]{Transformations preserving real-rootedness}

Several standard methods prove real-rootedness by applying a transformation
known to preserve real roots.  Coefficientwise methods based on
\hyperref[polyaFrequency]{Pólya frequency sequences} and
\hyperref[totallyNonnegative]{totally non-negative matrices} are collected on
their own page; the main examples there include
\hyperref[aissenSchoenbergWhitney]{Aissen--Schoenberg--Whitney},
\hyperref[edreiThomaTheorem]{Edrei--Thoma theory},
\hyperref[veroneseSectionsPF]{Veronese sections}, and
\hyperref[tnnPathMatrix]{path matrices}.  This page records two other
frequently used classes of transformations: basis transformations such as the
Narayana and Eulerian transformations, and coefficientwise products such as
\hyperref[hadamardProduct]{Hadamard products}.


\subsection[narayanaTransformations]{Narayana transformations and matrix products}

\name{Jianxi Mao} and \name{Lijie Wang} study a basis transformation
interpolating between the type $B$ and type $A$ Narayana polynomials
\cite{MaoWang2026x}.  For $m\in \setZ_{\geq 0}$, set
\[
  N_{n,m}(x)
  \coloneqq {}_2F_1(-n,-n-m;m+1;x)
  =
  \sum_{k=0}^n
  \frac{(-n)_k(-n-m)_k}{(m+1)_k\,k!}x^k,
\]
where $(a)_k$ is the Pochhammer symbol.  Then $N_{n,0}(x)$ is the
type $B$ Narayana polynomial and $N_{n,1}(x)$ is the type $A$
Narayana polynomial.  The corresponding
\defin{Narayana transformation}\label{narayanaTransformationDefinition}
$T_{N_m}$ is defined by
\[
  T_{N_m}(x^n)=N_{n,m}(x).
\]

\begin{theorem}[Mao--Wang, \cite[Thm.~1.1]{MaoWang2026x}]
\label{thm:narayanaTransformationMaoWang}
Let $m\in\setZ_{\geq 0}$.  Suppose that
\[
  p(x)=\sum_{k=0}^n a_kx^k \in \setR_{\geq 0}[x]
\]
has only real roots and has degree $n$.  Then
\[
  T_{N_m}(p(x))=\sum_{k=0}^n a_kN_{k,m}(x)
\]
has only real nonpositive roots.
\end{theorem}

In particular, the type $A$ and type $B$ Narayana transformations
preserve real-rootedness for polynomials with non-negative coefficients.
The proof uses rectangular additive convolution, and the non-negativity
assumption on the coefficients is essential.
This should be compared with the
\defin{Eulerian transformation}\label{eulerianTransformation}
\(A(x^n)=A_n(x)\).  Brenti conjectured that the Eulerian transformation
preserves real-rootedness, but this is false in full generality by
\name{Petter Brändén} and \name{Katharina Jochemko}
\cite{BrandenJochemko2022}.  A positive result in this direction is due to
\name{Christos Athanasiadis}, who proves real-rootedness for a large
non-negative cone; see the \hyperref[gammaPositivity]{gamma-positivity page}
and \cite{Athanasiadis2025EulerianTransformation}.

\begin{theorem}[Brenti, \cite[Thm.~2.4.2]{Brenti1995}]
\label{thm:brentiFallingFactorialBasis}
Let
\[
  \langle x\rangle_k\coloneqq x(x-1)\dotsm(x-k+1),
  \qquad \langle x\rangle_0\coloneqq 1.
\]
If
\[
  f(x)=\sum_{k=0}^n a_k\langle x\rangle_k
\]
has only real nonpositive roots, then
\[
  \sum_{k=0}^n a_kx^k
\]
has only real nonpositive roots.
\end{theorem}

Mao and Wang use this theorem as Lemma~3.2 of \cite{MaoWang2026x} in
their study of products of lower triangular matrices.  If
$M=(M(n,k))_{n,k\geq 0}$ is lower triangular, write
\[
  M_n(x)\coloneqq \sum_{k=0}^n M(n,k)x^k
\]
for its \defin{row-generating function}\label{rowGeneratingFunction} (RGF).
Their product criterion gives a broad class of matrices $B$ for which
real-rootedness of all $M_n(x)$ forces real-rootedness of the RGFs of
$MB^r$ for every $r\geq 1$ \cite[Thm.~1.3]{MaoWang2026x}.  This applies
when $B$ is the coefficient matrix of the polynomials $N_{n,m}(x)$, and
also to recurrence-defined triangular matrices including Pascal's
triangle and the Stirling triangles.

\begin{conjecture}[Mao--Wang, \cite[Conj.~4.1]{MaoWang2026x}]
\label{conj:eulerianDelannoySquares}
Let $A$ be the Eulerian triangle and let $D$ be the Delannoy triangle.
The row-generating functions of $A^2$ and $D^2$ have only real
nonpositive roots.
\end{conjecture}



\section[hadamardProduct]{Coefficientwise products and multiplier sequences}

Coefficientwise products are another standard way to preserve real roots.
There are three closely related forms in common use: finite
\hyperref[hadamardProductTheorems]{Hadamard product theorems}, infinite
diagonal operators or \hyperref[multiplierSequences]{multiplier sequences},
and normalized products such as the
\hyperref[schurSzegoComposition]{Schur--Szeg\H{o} composition} and
\hyperref[diamondProduct]{diamond product}.

The \defin{Hadamard product}\label{hadamardProductDefinition}
(or \emph{coefficientwise product})
of two polynomials $f(t)=\sum a_j t^j$ and $g(t)=\sum b_j t^j$ is
\[
  f \odot g \coloneqq \sum_j a_j b_j \, t^j.
\]

\subsection[multiplierSequences]{Multiplier sequences}

\begin{definition}[Multiplier sequence]
A sequence $\gamma=(\gamma_0,\gamma_1,\dotsc)$ is a
\defin{multiplier sequence}\label{multiplierSequenceDefinition}
if the diagonal operator
\[
  T_\gamma\!\left(\sum_{k\geq 0} a_k x^k\right)
  \coloneqq
  \sum_{k\geq 0} \gamma_k a_k x^k
\]
sends every real-rooted polynomial to a real-rooted polynomial or to zero.
\end{definition}

\begin{theorem}[Pólya--Schur, see \cite{SchurPolya1914,Karlin1968}]
A real sequence $\gamma=(\gamma_0,\gamma_1,\dotsc)$ is a multiplier
sequence if and only if the exponential generating function
\[
  \Phi_\gamma(x)=\sum_{k\geq 0}\gamma_k\frac{x^k}{k!}
\]
or $\Phi_\gamma(-x)$ has the form
\[
  Cx^m e^{ax}\prod_{j\geq 1}(1+\alpha_jx),
\]
where $C\in\setR$, $m\geq 0$, $a\geq 0$, $\alpha_j\geq 0$, and
$\sum_j\alpha_j<\infty$.
\end{theorem}

\subsection[hadamardProductTheorems]{Hadamard product theorems}

\begin{theorem}[Schur--Pólya; see {\cite[Thm.~3.4]{Wagner1992}}]
If $f$ and $g$ are real-rooted with non-negative coefficients,
then $f\odot g$ is also real-rooted with non-negative coefficients.
\end{theorem}

\begin{theorem}[Garloff--Wagner, see \cite[Thm.~4]{GarloffWagner1996}]
Let $f,g,p,q \in \setR[t]$ have non-negative coefficients and only
real non-positive zeros.  If
\[
  f \interl g
  \qquad\text{and}\qquad
  p \interl q,
\]
then
\[
  f\odot p \interl g\odot q.
\]
Here we use the zero convention for interleaving, so the conclusion also
covers the case where one of the Hadamard products is zero.
\end{theorem}

\begin{theorem}[Schur, see {\cite{Schur1914} and \cite[Thm.~1]{Branden2004operators}}]
Let
\[
  f(x)=a_0+a_1x+\dotsb+a_nx^n,
  \qquad
  g(x)=b_0+b_1x+\dotsb+b_mx^m
\]
be real-rooted polynomials.  Suppose that all zeros of $g$ have the
same sign.  Then
\[
  \sum_{k\geq 0} k!\,a_kb_kx^k
\]
is real-rooted, where missing coefficients are interpreted as zero.
If $a_0b_0\neq 0$, then all zeros are distinct.
\end{theorem}

\name{Jang Soo Kim} and \name{Jaeseong Oh} study Hadamard products of
Jacobi--Trudi matrices in relation to
\hyperref[totallyNonnegative]{total positivity} \cite{KimOh2025x}.
They give Schur-positive results for ribbon-indexed Hadamard squares; the
arXiv version notes that an attempted bijection has a flaw, so Sokal's
conjecture on Hadamard squares remains open.

\begin{theorem}[Laguerre, see \cite[Thm.~12]{Branden2004operators}]
If
\[
  f(x)=a_0+a_1x+a_2x^2+\dotsb+a_nx^n
\]
is real-rooted, then
\[
  a_0+a_1x+\frac{a_2}{2!}x^2+\dotsb+\frac{a_n}{n!}x^n
\]
is real-rooted.
\end{theorem}

The Schur--Pólya theorem follows from the fact that the class of
polynomials with only real non-positive zeros and non-negative
coefficients is a cone of multiplier sequences, closed under the
Hadamard product.  See \cite{Karlin1968} or the exposition in
\cite{Branden2015}.

\begin{example*}[Type $B$ Narayana polynomials via Hadamard product]
The type $B$ Narayana polynomials $\sum_{k=0}^n \binom{n}{k}^2 t^k$
are the Hadamard product of $(1+t)^n$ with itself.
Since $(1+t)^n$ is real-rooted, the Hadamard product theorem
immediately gives real-rootedness.
\end{example*}

The Hadamard product theorem is useful whenever a polynomial's
coefficients factor as a product of two sequences that are each
the coefficient sequence of a real-rooted polynomial.

\subsection[schurSzegoComposition]{Schur--Szeg\H{o} composition}

\begin{theorem}[Schur--Szeg\H{o} composition theorem, see \cite{RahmanSchmeisser2002}]
Let
\[
  F(x)=\sum_{k=0}^n \binom{n}{k} a_k x^k
  \qquad\text{and}\qquad
  G(x)=\sum_{k=0}^n \binom{n}{k} b_k x^k.
\]
Their \defin{Schur--Szeg\H{o} composition}\label{schurSzegoCompositionDefinition}
is
\[
  (F*G)(x)
  \coloneqq
  \sum_{k=0}^n \binom{n}{k} a_k b_k x^k.
\]
If $F$ and $G$ have only real non-positive zeros, then $F*G$ also has
only real non-positive zeros.
\end{theorem}

\subsection[diamondProduct]{Diamond product}

\begin{theorem}[Wagner--Brändén, {\cite[Thm.~2 and Thm.~13]{Branden2004operators}}]
For polynomials $f,h\in \setR[x]$, define the
\defin{diamond product}\label{diamondProductDefinition} by
\[
  f\diamond h
  \coloneqq
  \sum_{n\geq 0}
  \frac{f^{(n)}(x)h^{(n)}(x)}{n!\,n!}x^n(x+1)^n.
\]
If $h$ has all its zeros in the interval $[-1,0]$ and $f$ is
real-rooted, then $f\diamond h$ is real-rooted.  Moreover, if
$g \interl f$, then
\[
  g\diamond h \interl f\diamond h.
\]
In the special case where both $f$ and $h$ have all zeros in $[-1,0]$,
the polynomial $f\diamond h$ also has all zeros in $[-1,0]$.
\end{theorem}



\section[posetRealRooted]{Poset operations preserving real-rootedness}

\subsection[neggersStanleyConjecture]{Neggers--Stanley conjecture}

For a naturally labeled poset $P$, the
\hyperref[pEulerianPolynomial]{$P$-Eulerian polynomial} is the descent
generating polynomial of the linear extensions of $P$.  The
\defin{Neggers--Stanley conjecture}\label{neggersStanleyConjectureDefinition}
predicted that these polynomials have only
real zeros.  The conjecture is false in this generality: counterexamples were
constructed by \name{John R. Stembridge}
\cite{Stembridge2007NeggersStanleyCounter}.  Positive results for restricted
classes of posets are still important, especially because $P$-Eulerian
polynomials are $h^*$-polynomials of \hyperref[orderPolytope]{order
polytopes}.

\begin{example*}[P-Eulerian polynomials for series-parallel posets]
Let $E(P,t)$ denote the Eulerian polynomial associated with the naturally
labeled poset $P$.
The \defin{ordinal sum}\label{ordinalSumPoset} $P \oplus Q$ of two posets is
the poset
where
\[
x \prec y \iff x \prec_P y \text{ or } x \prec_Q y \text{ or } x \in P, y \in Q.
\]
The Eulerian polynomial then has the property that
\[
E(P \oplus Q, t) = E(P,t) \cdot E(Q,t).
\]
Moreover, for the \defin{disjoint union}\label{disjointUnionPoset} of posets,
we have
\[
E(P \sqcup Q, t) = E(P,t) \diamond E(Q,t),
\]
where $\diamond$ is the \hyperref[diamondProduct]{diamond product} above.
These operations preserve real-rootedness, so all series-parallel posets have
real-rooted Eulerian polynomials,
see \cite{Wagner1992,Wagner1992en}.
This result extends to labeled posets, see \cite{Branden2004operators}.
\end{example*}

\subsection[generalizedSnakeRealRooted]{Generalized snake posets}

\name{Benjamin Braun} and \name{Aryaman Jal} prove
\(h^*\)-real-rootedness for \hyperref[orderPolytope]{order polytopes} of
\defin{generalized snake posets}\label{generalizedSnakePoset}
\cite{BraunJal2026x}.  If \(w\) is a
generalized snake word and \(P(w)\) is the corresponding naturally labeled
width-two poset, then Stanley's theorem and the width-two poset model of
\name{Per Alexandersson} and \name{Aryaman Jal} identify
\[
  h^*(\mathcal{O}(P(w));t)
  =
  W_{P(w)}(t)
  =
  M_w(t),
\]
where \(W_{P(w)}(t)\) is the descent-generating polynomial of the linear
extensions of \(P(w)\), and \(M_w(t)\) is the corresponding non-nesting rook
polynomial.

\begin{theorem}[Braun--Jal \cite[Thm.~4.1]{BraunJal2026x}]
Let \(w\) be a word in \(L\) and \(R\) of positive length, and let \(w'\) be
obtained by deleting the last letter of \(w\).  Then \(M_{\epsilon w}(t)\) is real-rooted
and
\[
  M_{\epsilon w'}(t) \interl M_{\epsilon w}(t).
\]
\end{theorem}

The proof is a clean interlacing induction using a recurrence for the
non-nesting rook polynomials.  The constant words recover the Narayana
polynomials, while the regular snake words recover the
\defin{Delannoy row polynomials}\label{delannoyRowPolynomials} \(P_n(t)\),
whose coefficients are the antidiagonal rows of \oeis{A008288}.  These
polynomials satisfy
\[
  P_0(t)=1,\qquad P_1(t)=1+t,\qquad
  P_n(t)=(1+t)P_{n-1}(t)+tP_{n-2}(t) \quad (n\geq 2).
\]
Thus the Delannoy row polynomials are \(h^*\)-polynomials of order polytopes
of regular snake posets.  This is a different family from the zig-zag
Eulerian polynomials \(Z_m(t)\) discussed on the
\hyperref[gammaPositivity]{gamma-positivity page}; those polynomials are known
to be gamma-positive, but their real-rootedness is a separate open problem.
In particular, the first nontrivial regular-snake base case is
\[
  M_{\epsilon R}(t)=M_{\epsilon L}(t)=1+3t+t^2.
\]
This is also the first nontrivial row \(1,3,1\) of the Delannoy triangle
\oeis{A008288}, and the same polynomial appears in the
\hyperref[transferMatrixWeightedTilings]{weighted transfer-matrix tiling}
example.



\section[realRootedNormality]{Normality in the limit}

If $P_n(x)$ is a sequence of real-rooted polynomials with non-negative coefficients,
the coefficients often become asymptotically normally distributed.
The following theorem makes this precise.

\begin{theorem*}[Bender 1973, Harper 1967]
Suppose
\[
P_n(x) = \sum_{k} a_n(k) x^k = a_n \prod_j (x + r_j)
\]
where $r_j \geq 0$. Define
\[
\mu_n = \sum_j \frac{1}{1 + r_j}, \quad \sigma_n^2 = \sum_j \frac{r_j}{(1 + r_j)^2}.
\]
If $\sigma_n \to \infty$, then the \emph{normalized coefficients}
\[
p_n(k) = \frac{a_n(k)}{P_n(1)}
\]
are \emph{asymptotically normal}: if $X_n$ is the random variable
with distribution $p_n(k)$, then
\[
\frac{X_n - \mu_n}{\sigma_n} \xrightarrow{d} \mathcal{N}(0,1).
\]
Moreover, if the $a_n(k)$ are log-concave (which follows from real-rootedness), a local limit theorem holds:
\[
p_n(k) \approx \frac{1}{\sigma_n \sqrt{2 \pi}} e^{ - (k - \mu_n)^2 / (2 \sigma_n^2) }
\]
uniformly in $k$.
\end{theorem*}

See \cite{Bender1973} for the original result and examples including ordered set partitions,
\hyperref[ex:eulerianSturm]{Eulerian numbers},
and \hyperref[realRootedMatching]{matchings on the $2\times n$ grid graph}.
For Baxter permutations, see \cite{Zhao2024x}.
For P-recursively defined polynomials, see \cite{Li2025x}.
Several further examples are provided in \cite{Hitczenko2024x}.
